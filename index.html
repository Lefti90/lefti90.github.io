DemoLab - Kevät 2020

1. Johdanto

Tässä dokumentissa käydään läpi DemoLabin aikana tekemiäni tuotoksia liittyen pelin ohjelmointiin ja osittain myös sisällöntuotantoon. Peliprojekti, jossa olin mukana, kantaa nimeä Devoid Tomorrow. Keskityn eniten niihin asioihin, missä minulla on ollut isoin rooli tai jotka olen toteuttanut kokonaan itse. Tämän DemoLabin aikana työnsin nenääni moneenkin asiaan, mutta jos en ole varsinaisesti toteuttanut lopulliseen tulokseen vaikuttavia suorituksia, jätän ne huomiotta.

Mainittakoon tässä alussa kuitenkin se, että kevätlukukauden alussa, en ollut tehnyt Unityn kanssa oikeastaan mitään melkein vuoteen ja aloitimme aluksi toisen ohjelmoijan, Aleksi Rajamäen, kanssa herättelemällä vanhoja taitoja pariohjelmoijalla, jolla saimme molemmat hyvän startin projektille, mutta oikeastaan ensimmäisen sprintin jälkeen jaoimme molemmille omat osa-alueet, joihin keskityimme.

Kurssin aikana olemme kuitenkin kyselleet ja autelleet toisiamme, jos on tullut jotain askarruttavia haasteita. Ohjelmoinnissa on ollut myös myöhemmin kuvaan astunut kolmas ohjelmoija, Henri Kinnunen, jonka osaamista on myös hyödynnetty hänen omissa osa-alueissaan, sekä koodareiden keskenäisessä vuorovaikutuksessa syntynyttä keskenäistä auttamista ja ongelman ratkontaa.

DemoLabin aikana halusin oppia lisää ohjelmoinnista, sekä pelien näkökulmasta, että yleisesti ohjelmointiin liittyvistä haasteista ja niiden ratkaisemisesta. Halusin oppia tekemään toimivaa, selkeää ja tehokasta koodia, mikä toimisi mahdollisimman universaailsti, ilman kovakoodia tai niin kutsuttua ”spagettia”.

Projektin omia osa-alueitani tutkiessa, voin huomata, että olen kehittynyt ohjelmoijana ja olen edistynyt tavoitteissani. Oppimistavoitteiden lisäksi olen oppinut asioita mm. ryhmätyöskentelystä, projektityöskentelystä, ongelman ratkonnasta ja etätyöskentelystä.

4.1. Dialogi

Projektin alussa pariohjelmointi toimi hetken, mutta kun molemmat olimme päässeet vauhtiin, se ollut enää tehokasta tai järkevää. Täten jaoimme omat osa-alueet ja tein hetken aikaa peliin UI:n liittyviä asioita.

Tämän jälkeen PO:mme Ari-Pekka Savoilainen sanoi, että voisin ruveta kehittämään järjestelmää, millä tarinaa vietäisiin eteenpäin. Hän oli löytänyt netistä Unityn kanssa toimivan dailogi-pluginin mihin tarinan kirjoittaminen olisi helppoa ja sen voisi liittää suoraan unityyn.
En enää muista ohjelman nimeä, mutta nopeasti kävi ilmi, että unityn puolella tämä ko. plugin oli aika kömpelö.
Onneksi naapuritoimiston Niko Tervo oli löytänyt toisen pluginin nimeltään YarnSpinner (jatkossa Yarn), mitä hän käytti projektissaan Senpai Garden.
Niko antoikin hyviä neuvoja, jotta pääsin alkuun Yarnin kanssa.

YarnSpinner on siis oma scriptauskielensä ja se pystyy kommunikoimaan Unityn kanssa. Hieman kömpelön syntaxin takia, Yarnin opettelu vei hieman aikaa, mutta lopulta se on aika yksinkertainen kieli.

Suurin osa työstä Yarnin kanssa olikin opettelua, sekä "CustomCommand":en tekoa.

Yarnin "YarnCommands":it toimivat seuraavasti <<komento objekti string>>.
Esimerkki:
<<showFace thug2 Left>> Eli aluksi "komento" haetaan "YarnCommands.cs":tiedostosta esim. "showFace".
Sitten toisena "objekti", eli mille objektille halutaan tehdä jotain esim. "thug2"
Viimeisenä Yarn haluaa aina passata stringin, vaikka sitä ei tarvittaisi, mutta tässä tilanteessa "Left" on olennainen osa komentoa.
Tässä esimerkissä "showFace" näyttää "thug2" nimisen naaman dialogin vasemmassa "Left" reunassa.
Alla esimerkki Yarnista ja siihen tehdyistä komennoista. Scriptien alussa on kommentoitu, mitä ko. scripti tekee.

//Tänne tulevat kaikki YarnSpinnerin itse tehdyt "CustomCommand":it
public class YarnCommands : MonoBehaviour
{
	//taustakuvan liikutus
	string activateNext;
	private TargetManagerScript targetManager;
	private SetNewTarget setNewTarget;

	//naamakuvan liikutus
	[SerializeField] private moveImageScript mis;
	//naamat esiin ja piiloon
	public GameObject faceImage;
	public RawImage image;
	//scenen vaihto
	private void Awake()
	{
		Cursor.visible = true;
		Cursor.lockState = CursorLockMode.None;
		targetManager = GameObject.FindObjectOfType();
		setNewTarget = GameObject.FindObjectOfType();
	}
	//vaihdetaan taustakuva
	[YarnCommand("bg_Change")]
	public void setActive1(string bg_Change)
	{
		Debug.Log("bg_Change: " + bg_Change);
		activateNext = "bg_Change";
		//lähetetään setNewTargetille uusi target
		setNewTarget.ChangeTarget(bg_Change);
	}
	//näytetään naama
	[YarnCommand("showFace")]
	public void showFaceImage(string position)
	{
		mis = null;
		//naaman Imagen alpha ykköseksi
		Debug.Log("showFace "+position);
		image = faceImage.GetComponent();
		Debug.Log("AAAAAAAAAAAAAAAAAAAA: " + image.name);
		var tempColor = image.color;
		tempColor.a = 1;
		image.color = tempColor;
		//lähetetään moveImageScriptille position ja itse objekti
		mis = GameObject.FindObjectOfType();
		mis.moveImage(position, faceImage);
	}
	//piilotetaan naama
	[YarnCommand("hideFace")]
	public void hideFaceImage(string position)
	{
		//naaman Imagen alpha nollaksi
		//Debug.Log("hideFace: "+position);
		image = faceImage.GetComponent();
		var tempColor = image.color;
		tempColor.a = 0;
		image.color = tempColor;
	}
	//Vaihdetaan sceneä
	[YarnCommand("changeScene")]
	public void changeScene(string scene)
	{
		ShangeScenes shangeScenes;
		shangeScenes = GameObject.FindObjectOfType();
		shangeScenes.SceneShanging(scene);
	}
}

//Tämä scripti vastaa naamojen kuvien siirtelystä
//Kuvan paikat ja muut tulevat "YarnCommands.cs":stä
public class moveImageScript : MonoBehaviour
{
	[SerializeField] private Transform facePosLeft;
	[SerializeField] private Transform facePosCenter;
	[SerializeField] private Transform facePosRight;
	[SerializeField] private Transform facePosHidden;
	[SerializeField] private GameObject _imageObj;
	[SerializeField] private string newpos;
	[SerializeField] private string currentPos;


	/// 
	/// Dialogin naamojen siirtäminen
	/// 
	///  Vastaanotetaan string: YarnCommands.cs 
	///  Vastaanotetaan GameObject: YarnCommand.cs 
	public void moveImage(string position, GameObject imageObject)
	{
		var posTaken = GameObject.FindGameObjectWithTag("Face" + position);
		if (posTaken != null)
		{
			//siirretään vanha kuva hiddeniin
			posTaken.transform.position = facePosHidden.position;
			posTaken.tag = "FaceHidden";
		}
		else
		{
			Debug.Log("error in moveImageScript");
		}
		//lopuksi siirretään kuva paikalleen
		newpos = position;
		_imageObj = imageObject;
		currentPos = position;
		imageObject.tag = "Face" + position;
		Debug.Log("set position: " + position);
		#region Positions
		if (position == "Center")
		{
			imageObject.transform.position = facePosCenter.position;
		}
		else if (position == "Left")
		{
			imageObject.transform.position = facePosLeft.position;
		}
		else if (position == "Right")
		{

			imageObject.transform.position = facePosRight.position;
		}
		else if (position == "Hidden")
		{
			imageObject.transform.position = facePosHidden.position;
		}
		else
		{
			Debug.Log("Error in face position");
		}
		#endregion
	}
}

4.2. Inventory

Yksi isoimmista ja haastavimmista asioista projektin suhteen minulle oli "Inventoryn" ja siihen liittyvien asioiden kehittäminen.

No mikä on inventory?

Se on esinevarasto, mihin pelaajan keräämät esineet varastoidaan.

Kuulostaa helpolta, mutta tähän liittyy paljon erilaisia muuttujia.

Ensimmäinen ongelma: Päätin tehdä inventoryn "ScriptableObject:lla" (jatkossa SO). En ollut ikinä ennen kuullutkaan niistä, mutta Henri ehdotti sitä ja se vaikutti järkevältä, mitä se lopuksi olikin.

Aluksi minun piti opetella mitä SO:t ovat ja miten niitä käytetään. Lähdin tekemään ensimmäistä versioita suoraan sitä mukaan mitä opin, mutta koska olin ymmärtänyt asioita väärin, jouduin aloittamaan alusta.

Seuraavan version ajattelin aloittaa tekemällä oman tyhjän projektin missä voin harjoitella SO:den käyttämistä. Tämä oli hyvä ja seurasin ohjeita tutoriaalista. Tutoriaali oli kuitenkin niin monimutkainen, että kun luulin tajunneeni, kuinka voin soveltaa tutoriaalin oppeja projektiin, meni sormi taas suuhun.

Kolmannen kerran ajattelin, että nyt en katso mistään mallia ja teen niin yksinkertaisen version kun vain osaan. Tämä koituikin hyväksi taktiikaksi, kunnes seinä tuli taas vastaan. Onneksi olin pitänyt projektin niin yksinkertaisena, että kun pyysin Henriltä apua, hän osasi suoraan neuvoa miten voisin jalostaa projektia eteenpäin.


Kolmas versio alkoikin olla toimiva, jonka integroin oikeaan projektiin.

Vaikka integrointi oikeaan projektiin ei tapahtunutkaan aivan nappia painamalla, ei siinä tullut suurempia haasteita vastaan.
Yhdistämisen jälkeen aloin päivittämään ulkoasua projektin tyylin mukaiseksi, tämä olikin hyvin paljon vain kuvien vaihtamista sopiviin.

//Aseen SO:ssa käsitellään Partit, mitä voidaan yhdistää aseeseen
//Scripti katsoo mitä Part:ia ollaan laittamassa ja yhdistää sen aseen oikeaan kohtaan ja
//ottaa mukanaan statsit ja muut tiedot, mitä se sisältää
[CreateAssetMenu(fileName = "Weapon", menuName = "Weapon")]
public class WeaponSO : ScriptableObject
{
	public new string name;
	public ItemType partID;
	public Part barrel;
	public Part handguard;
	public Part body;
	public Part stock;
	public Part handle;
	public Part mag;
	public Part sights;

	[Space, Header("Stats")]
	public float ROF;
	public float DMG;
	public float ACC;
	public float MAX_AMMO;

	public void EquipPart(Part partToEquip)
	{
		switch (partToEquip.partID)
		{
			case ItemType.Barrel:
				if (barrel != null)
					barrel.equipped = false;
				barrel = partToEquip;
				barrel.equipped = true;
				break;
			case ItemType.HandGuard:
				if (handguard != null)
					handguard.equipped = false;
				handguard = partToEquip;
				handguard.equipped = true;
				break;
			case ItemType.Body:
				if (body != null)
					body.equipped = false;
				body = partToEquip;
				body.equipped = true;
				break;
			case ItemType.Stock:
				if (stock != null)
					stock.equipped = false;
				stock = partToEquip;
				stock.equipped = true;
				break;
			case ItemType.Handle:
				if (handle != null)
					handle.equipped = false;
				handle = partToEquip;
				handle.equipped = true;
				break;
			case ItemType.Magazine:
				if (mag != null)
					mag.equipped = false;
				mag = partToEquip;
				mag.equipped = true;
				break;
			case ItemType.Sights:
				if (sights != null)
					sights.equipped = false;
				sights = partToEquip;
				sights.equipped = true;
				break;
			default:
				Debug.Log("Error finding the button");
				break;
		}
		UpdateWeaponStats();
	}
	public void UpdateWeaponStats()
	{
		//Yhdistetään statsit
		ROF = barrel.ROF + handguard.ROF + body.ROF + stock.ROF + handle.ROF + mag.ROF + sights.ROF;

		DMG = barrel.DMG + handguard.DMG + body.DMG + stock.DMG + handle.DMG + mag.DMG + sights.DMG;

		ACC = barrel.ACC + handguard.ACC + body.ACC + stock.ACC + handle.ACC + mag.ACC + sights.ACC;

		MAX_AMMO = barrel.MAX_AMMO + handguard.MAX_AMMO + body.MAX_AMMO + stock.MAX_AMMO + handle.MAX_AMMO + mag.MAX_AMMO + sights.MAX_AMMO;
	}
}

			//Tyhjätään lista
			foreach (Button button in panelToAttachButtonsTo.transform.GetComponentsInChildren<Button>())
			{
				Destroy(button.gameObject);
			}
			displayedItems.Clear();
	
			//Täytetään lista ItemTypen mukaan

			{            
				if (Item.partID == (ItemType)itemSlotID)

				}
				else
				Debug.Log("ItemSlot error");

4.3. PauseMenu

PauseMenun on tehty pääosin UI:elementeistä, mutta mukaan haluttiin myös 3D-objekteja ja 3D-efektejä
Tämä aiheutti sen, että Canvasta ei voinut pitää normaalina Overlay-elementtinä vaan menun piti ottaa huomioon kameran kuvakulma.
PauseMenu on toteutettu kahdella eri kameralla, toinen on jossain maailman ulkopuolella ja menu on oikeasti sen kameran edessä.
Tämän jälkeen toisen kameran kuvaama kuva heijastetaan pääkameran eteen, kun peli laitetaan Pauselle.

Alunperin minun ei pitänyt tehdä menujen grafiikkaa, mutta se päätyi lopulta minun syliini, mikä on sinänsä ihan kiva, koska tykkään kuvankäsittelystä.

//Nappuloiden animaatiot PauseMenussa
public class ButtonHilightScript : MonoBehaviour, IPointerEnterHandler, IPointerExitHandler
{
    public GameObject nuoli;
    public GameObject nappula;
    public Image orgImg;
    public Sprite orgSprite;
    public Sprite HilightSprite;
    private Vector2 hilightShadow = new Vector2(-19f, -18f);
    private Vector2 normalShadow = new Vector2(-12f, -12f);

    private void Awake()
    {
        nappula = this.gameObject;
        orgImg = this.gameObject.GetComponent();
        orgSprite = orgImg.sprite;
        nuoli.SetActive(false);
    }

    public void OnPointerEnter(PointerEventData eventData)
    {
        nuoli.SetActive(true);
        nuoli.transform.LeanSetLocalPosY(this.transform.localPosition.y + 20f);
        LeanTween.moveLocalX(nappula, 50f, 0.1f).setIgnoreTimeScale(true);
        nappula.gameObject.GetComponent().overrideSprite = HilightSprite;
        nappula.gameObject.GetComponent().effectDistance = hilightShadow;
    }

    public void OnPointerExit(PointerEventData eventData)
    {
        nuoli.SetActive(false);
        LeanTween.moveLocalX(nappula, 0f, 0.1f).setIgnoreTimeScale(true);
        nappula.gameObject.GetComponent().overrideSprite = orgSprite;
        nappula.gameObject.GetComponent().effectDistance = normalShadow;
    }
}

		rangedWMenu.SetActive(false);

	// Update is called once per frame
	void Update()
	{
		if (Input.GetKeyDown(KeyCode.Escape))
		{
			if (pauseMenuUi.activeSelf == true && pauseMenuUi.transform.position == activePos.transform.position)//GamePaused == true && OtherMenuOpen == false)
			{
				Debug.Log("Resume");
				Resume();
			}
			else if (pauseMenuUi.activeSelf == false)//(GamePaused == false && OtherMenuOpen == false)
			{
				Debug.Log("Pause");
				Pause();
			}
			else if (GamePaused == true && OtherMenuOpen == true)
			{
				Debug.Log("CloseCurrentMenu, Game paused: "+GamePaused+" OthermenuOpen: "+OtherMenuOpen);
				CloseCurrentMenu();
			}
			else
			{
				Debug.Log("Else");
				Debug.Log(GamePaused);
				Debug.Log(OtherMenuOpen);
				if (parentMenu != null)
				{
					parentMenu.tag = "DeactiveMenu";
					parentMenu = null;
				}
			}
		}
	}
